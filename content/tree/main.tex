\templateentry
{Base}
{Estrutura base com HLD: \texttt{addEdge} e \texttt{build(root)} em $O(n)$.}
{Base de árvore com parentesco, profundidade, tamanhos de subárvore e heavy-light order. Interface: construa \texttt{Tree t(n)}, adicione arestas com \texttt{addEdge(u, v)} e finalize com \texttt{build(root)}. Depois disso, campos como \texttt{par}, \texttt{dep}, \texttt{head}, \texttt{in}, \texttt{out} e \texttt{rev} ficam prontos para uso. Complexidade: $O(n)$.}
{tree/base.cpp}

\templateentry
{Utilidades}
{Helpers sobre \texttt{Tree}: ancestral, LCA, distância e navegação em caminhos; em geral $O(\log n)$.}
{Utilitários que assumem uma \texttt{Tree} já construída. Interface:
\begin{itemize}
	\item \texttt{isAnc(t, a, b)} testa se \texttt{a} é ancestral de \texttt{b}.
	\item \texttt{subtree(t, v)} devolve o intervalo Euler da subárvore.
	\item \texttt{lca(t, a, b)} e \texttt{dist(t, a, b)} fazem LCA e distância.
	\item \texttt{kthAnc(t, v, k)} sobe \texttt{k} ancestrais.
	\item \texttt{jump(t, a, b, k)} pega o $k$-ésimo vértice no caminho \texttt{a -> b}.
	\item \texttt{getPath(t, a, b, f)} percorre o caminho em segmentos Euler e chama \texttt{f(l, r)}.
\end{itemize}
Complexidade: \texttt{isAnc} e \texttt{subtree} são $O(1)$; \texttt{lca}, \texttt{dist}, \texttt{kthAnc} e \texttt{jump} são $O(\log n)$; \texttt{getPath} enumera $O(\log n)$ segmentos.}
{tree/helpers.cpp}

\templateentry
{LCA}
{\texttt{LCA lca(g, root)} e \texttt{lca.lca(a, b)}; build $O(n \log n)$, query $O(1)$.}
{LCA por Euler tour + sparse table. Interface: construa \texttt{LCA lca(g, root)} a partir da lista de adjacência e consulte pares com \texttt{lca.lca(a, b)}. Preprocessamento $O(n \log n)$, consulta $O(1)$.}
{tree/lca.cpp}

\templateentry
{Matching em Árvores}
{\texttt{treeMatching(g, root)} retorna o parceiro de cada vértice em $O(n)$.}
{Emparelhamento máximo em árvore por DFS gulosa em pós-ordem. Interface: passe a lista de adjacência 0-indexada e a raiz opcional; o retorno é um vetor \texttt{match} com o parceiro de cada vértice, ou \texttt{-1} se ele ficar livre. Complexidade: $O(n)$.}
{tree/tree-matching.cpp}
