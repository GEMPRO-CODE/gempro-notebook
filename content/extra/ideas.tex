\subsection{Calculando somas e integrais}
\begin{itemize}
	\item Verificar se existe uma fórmula fechada
	\item Inverter a ordem da soma (em somatórios duplos)
	\item Transformar em somatório duplo e inverter a ordem
	\item Verificar se a soma pode ser calculada de um jeito eficiente (sem ser uma fórmula fechada)
	\item No caso de calcular várias somas, verificar se existe uma relação entre elas (Calcular uma pode ajudar a calcular outra)
\end{itemize}
