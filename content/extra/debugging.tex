\subsection{Geral}
\begin{itemize}
	\item Trocou nomes de variáveis? ($m$ por $n$, $j$ por $i$, ...)
	\item Esqueceu algum caso especial? ($n = 1$, $m = 0$, ...)
	\item Esqueceu de trocar entre 0-based e 1-based?
\end{itemize}
\subsection{WA - Wrong Answer}
\begin{itemize}
	\item Esqueceu o \verb|define int i64|?
	\item Esqueceu algum caso base?
\end{itemize}
\subsection{RE - Runtime Error}
\begin{itemize}
	\item Verificar asserts.
	\item Verificar limites de vectors.
	\item Verificar se \verb|maxn| é o suficiente.
\end{itemize}
