\begin{itemize}[leftmargin=*,label={}]
	\item \textbf{Wrong Answer}
	\begin{itemize}
		\item reler o enunciado, os exemplos e as restrições.
		\item sua solução passa os exemplos?
		\item a resposta pede valor exato ou módulo?
		\item o módulo digitado está correto, e é o mesmo que o enunciado?
		\item verificar todas as operações aritméticas consideram o módulo, buscar todas as operações no código e verificar manualmente.
		\item verificar casos especiais, $n = 0$, $n = 1$, etc.
		\item testar manualmente todos os casos com input pequeno.
		\item considerou que $-x \mod y$ pode ser negativo?
		\item ver manualmente todos os índices de array e verificar se é possível que o índice esteja fora do array.
		\item limpou todas as variáveis globais em casos de teste?
		\item todas as variáveis são inicializadas corretamente?
		\item arrays são inicializados antes de serem usados?
	\end{itemize}

	\item \textbf{Runtime Error}
	\begin{itemize}
		\item ver manualmente todos os índices de array e verificar se é possível que o índice esteja fora do array.
		\item os índices vão até $n - 1$ ou até $n$? o array tem o tamanho certo? ($n + 1$ se os índices vão até $n$)
		\item os limites dos loops estão corretos? trocou $m$ e $n$?
		\item os arrays tem os tamanhos corretos? o tamanho era $2n$ em vez de $n$?
	\end{itemize}

	\item \textbf{Time Limit Exceeded}
	\begin{itemize}
		\item algum loop infinito acidental?
		\item recursão infinita? impediu de visitar o mesmo estado várias vezes?
		\item algum fator de $\log$ extra? talvez por causa de um set ou map.
		\item a complexidade amortizada está realmente correta?
		\item sem querer copiou objetos grandes várias vezes?
		\item inicializou/limpou objetos grandes sempre em cada caso de teste?
	\end{itemize}

	\item \textbf{Grafos e árvores.}
	\begin{itemize}
		\item self-loops e multi-edges importam?
		\item tratou o caso desconexo?
	\end{itemize}
\end{itemize}
