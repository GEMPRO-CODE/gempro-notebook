\extendedonly{
\subsubsection{Encontrando qualquer caminho euleriano}
Em grafos direcionados:
\begin{itemize}
	\item Se exatamente um vértice tem $d_{out} - d_{in} = 1$, então comece nesse vértice.
	\item Se todos os vértices tem $d_{out} = d_{in}$, pode começar em qualquer vértice com $d_{out} > 0$.
	\item Caso contrário, não existe caminho euleriano.
\end{itemize}
Em grafos não direcionados:
\begin{itemize}
	\item Se exatamente dois vértices tem $deg$ ímpar, então comece em qualquer um deles.
	\item Se todos os vértices tem $deg$ par, pode começar em qualquer vértice com $deg > 0$.
	\item Caso contrário, não existe caminho euleriano.
\end{itemize}
}

\subsection{Max Flow}
\subsubsection{Dinic}
\varianttext
{\texttt{addEdge(u, v, cap)} e \texttt{flow(s, t)} para fluxo máximo.}
{Dinic para fluxo máximo. Interface:
\begin{itemize}
	\item \texttt{addEdge(from, to, capacity)} adiciona uma aresta dirigida.
	\item \texttt{flow(s, t)} retorna o fluxo máximo de \texttt{s} para \texttt{t}.
	\item \texttt{flow(s, t, max)} faz o mesmo, mas limita o valor total a \texttt{max}.
\end{itemize}
Complexidade: $O(N^2M)$ no pior caso, e também limitada por $O(FM)$ quando o fluxo máximo é $F$.}

\subsection{Min Cost Flow}
\subsubsection{MCMF (SPFA)}
\varianttext
{\texttt{addEdge(u, v, cap, cost)} e \texttt{flow(s, t)} para min-cost max-flow.}
{Fluxo de custo mínimo com suporte a custos negativos. Interface:
\begin{itemize}
	\item \texttt{addEdge(from, to, cap, cost)} adiciona uma aresta com capacidade e custo por unidade.
	\item \texttt{flow(s, t)} retorna \texttt{\{max\_flow, min\_cost\}}.
\end{itemize}
Complexidade: $O(VE \cdot flow)$ no pior caso.}
